\newcommand{\operationcontract}[4]{
	\par
	\fbox{
		\begin{minipage}{\linewidth}
			\textbf{Operazione:} #1 \\
			\textbf{Riferimenti:} #2 \\
			\textbf{Pre-condizioni:} #3 \\
			\textbf{Post-condizioni:} #4 \\
		\end{minipage}
	} 
	\par
	\vspace{1\baselineskip}
}
